import yfinance as yf
import matplotlib.pyplot as plt
import seaborn as sns
import plotly.express as px
import pandas as pd
#Minha base de dados sobre a Meta da taxa SELIC veio do site oficial do Banco Central
#Minha base de dados sobre o Fed Fund Rate veio do site oficial do Federal Reserve Bank of St.Louis

# Especifique o par de moedas (BRL/USD) que desejo analisar
currency_pair = 'USDBRL=X'  # Taxa de cmbio BRL para USD

# Defina o intervalo de datas para os dados
start_date = '2004-01-01'
end_date = '2023-09-27'

# peguei os dados de taxa de cmbio do Yahoo Finance, nomeando os dados como 'exchange_rate_data'
exchange_rate_data = yf.download(currency_pair, start=start_date, end=end_date)



#Traando um grfico interativo onde 'exchange_rate_data.index' ser o eixo x, que neste caso a data, e a taxa de cmbio USD/BRL como eixo y.
fig = px.line(exchange_rate_data, x=exchange_rate_data.index, y='Adj Close',
              title='Value of Brazilian Real (BRL) Against USD Over Time',
              labels={'Adj Close': 'BRL/USD Exchange Rate'},
              template='plotly_dark')

# Customize hover tooltips
fig.update_traces(mode='lines+markers', hovertemplate='%{y:.2f} BRL/USD<br>%{x|%b %Y}')

# Mostre o grafico interativo
fig.show()

# Crie um lineplot Seaborn para o primeiro conjunto de dados, que  a coluna 'Diferencial' do conjunto de dados 'juntos', em que o eixo x  a data e o eixo y o valor do diferencial de juros
plt.figure(figsize=(10, 6))
ax = sns.lineplot(x='DateTime', y='Diferencial', data=juntos, label='Diferencial', marker='d', linestyle='-', color='red', linewidth=2)

#Crie um eixo y secundrio para o segundo conjunto de dados
ax2 = ax.twinx()

# Plote o segundo conjunto de dados no eixo y secundrio, que  o valor do USD/BRL que seria a coluna 'Fechar' do conjunto de dados 'juntos'
sns.lineplot(x='DateTime', y='Close', data=juntos, label='USD/BRL', marker='d', linestyle='-', color='purple', linewidth=2)

# Defina a cor do rtulo do primeiro eixo y para corresponder  cor da linha (vermelho)
ax.set_ylabel('Diferencial de Juros', color='red')

# Defina a cor do rtulo do segundo eixo y para corresponder  cor da linha (roxo)
ax2.set_ylabel('USD/BRL', color='purple')

#Adicione rtulos e ttulo
ax.set_xlabel('Data')
plt.title('Comparao entre diferencial de juros e USA')

# Adicionar legendas
lines1, labels1 = ax.get_legend_handles_labels()
lines2, labels2 = ax2.get_legend_handles_labels()
lines = lines1 + lines2
labels = labels1 + labels2
ax.legend(lines, labels, loc='upper left')

# Mostrar o grfico
plt.grid(True)
plt.show()
